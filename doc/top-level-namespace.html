post ("/login") do
    username=params[:username]
    password=params[:password]

    result = login(username, password)
    pwdigest = result["Password"]
    id = result["UserId"]

    if BCrypt::Password.new(pwdigest) == password
        session[:id] = id
        redirect("/my_site")
    else
        redirect("error_password")
    end
end
